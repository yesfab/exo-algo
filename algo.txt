tri_Insertion

PROCEDURE tri_Insertion ( Tableau t)
    n ← taille tableau t
    POUR i VARIANT DE 0 à n
        min ← i
        POUR j de i + 1 à n - 1
              si t[j] < t[min], alors min ← j
        FIN POUR
        si min ≠ i, alors échanger t[i] et t[min]
    FIN POUR
FIN PROCEDURE


tri_Selection

PROCEDURE tri_selection ( list a[1:n])
    POUR i VARIANT DE 1 A n - 1 FAIRE
        TROUVER [l] LE PLUS PETIT ELEMENT DE [i + 1:n];
        ECHANGER [l] ET [i];
FIN PROCEDURE;



tri_bulle ( TABLEAU)
    POUR i dans la taille du tableau
        pour u variant de 0 a taille du tableau
            si tableau [u+i] inferieur tableau[u]
            tableau[u+i] = tableau [u]
            tableau [u] = tableau[u+i]

continue <- VRAI

fin tant que
